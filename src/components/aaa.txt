import React, { useRef, useState } from 'react';

const Camara = ({ onCapture }) => {
  const webcamRef = useRef(null);
  const [imgSrc, setImgSrc] = useState(null);

  const capture = () => {
    const imageSrc = webcamRef.current.getScreenshot();
    setImgSrc(imageSrc);
    const hora = new Date().toLocaleString('es-ES', {
      dateStyle: 'full',
      timeStyle: 'medium'
    });
    onCapture(imageSrc, hora);
  };

  return (
    <div className="bg-white p-6 rounded-lg shadow-lg">
      <h2 className="text-xl font-bold mb-4">1. Capturar Evidencia Fotográfica</h2>
      
      <div className="flex flex-col md:flex-row gap-6">
        <div className="flex-1">
          <div className="border-2 border-gray-300 rounded-lg overflow-hidden">
            {imgSrc ? (
              <img src={imgSrc} alt="Captura" className="w-full h-auto" />
            ) : (
              <Webcam
                audio={false}
                ref={webcamRef}
                screenshotFormat="image/jpeg"
                className="w-full h-auto"
                videoConstraints={{
                  facingMode: 'environment',
                  width: { ideal: 1280 },
                  height: { ideal: 720 }
                }}
              />
            )}
          </div>
          
          <div className="mt-4 flex gap-4">
            <button
              onClick={capture}
              className="flex-1 bg-blue-600 text-white py-3 rounded-lg hover:bg-blue-700 font-medium"
            >
              {imgSrc ? 'Tomar Otra Foto' : 'Capturar Foto'}
            </button>
            
            {imgSrc && (
              <button
                onClick={() => setImgSrc(null)}
                className="flex-1 bg-gray-500 text-white py-3 rounded-lg hover:bg-gray-600"
              >
                Reintentar
              </button>
            )}
          </div>
        </div>
        
        <div className="flex-1 bg-gray-50 p-4 rounded-lg">
          <h3 className="font-bold mb-2">Instrucciones:</h3>

            <li>Mantenga el dispositivo estable</li>
          </ul>
        </div>
      </div>
    </div>

  const [respuestas, setRespuestas] = useState(datosPrevios || {});

  const marcar = (item, valor) => {
    const nuevo = { ...respuestas, [item]: valor };

    onChange(nuevo); // 👉 AQUÍ SE GUARDA EN APP

    <div>
      <h2 style={{ textAlign: 'center', marginBottom: 20 }}>Inspección de Unidad</h2>

              display: "flex", 
              justifyContent: "space-between",

              borderBottom: "1px solid #eee"

                      padding: "8px 12px",

import React from 'react';
import { jsPDF } from 'jspdf';

const Reporte = ({ datos }) => {

  // --- 1. LÓGICA DE UBICACIÓN E ID ---
  const generarID = () => {
    // Convertimos a minúsculas y quitamos acentos para asegurar detección
    const textoUbicacion = (datos.direccion || "").toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
    
    let codigo = "SW000"; // Por defecto

    // Prioridad a oficinas divisionales
    if (textoUbicacion.includes("montejo") || textoUbicacion.includes("alvarado") || textoUbicacion.includes("divisionales")) {
      codigo = "SW000";
    } 
    // Ciudades
    else if (textoUbicacion.includes("merida")) { codigo = "SW010"; }
    else if (textoUbicacion.includes("motul")) { codigo = "SW020"; }
    else if (textoUbicacion.includes("ticul")) { codigo = "SW030"; }
    else if (textoUbicacion.includes("campeche")) { codigo = "SW040"; }
    else if (textoUbicacion.includes("carmen")) { codigo = "SW050"; }
    else if (textoUbicacion.includes("chetumal")) { codigo = "SW060"; }
    else if (textoUbicacion.includes("tizimin")) { codigo = "SW070"; }
    else if (textoUbicacion.includes("cancun")) { codigo = "SW120"; }
    else if (textoUbicacion.includes("riviera") || textoUbicacion.includes("maya")) { codigo = "SW200"; }

    // Fecha y Hora exacta (DDMMAAAAHHMMSS)
    const now = new Date();
    const dia = String(now.getDate()).padStart(2, '0');
    const mes = String(now.getMonth() + 1).padStart(2, '0');
    const anio = now.getFullYear();
    const hora = String(now.getHours()).padStart(2, '0');
    const min = String(now.getMinutes()).padStart(2, '0');
    const seg = String(now.getSeconds()).padStart(2, '0');

    return `${codigo}${dia}${mes}${anio}${hora}${min}${seg}`;
  };

  const generarPDF = async () => {
    try {
      const ID_GENERADO = generarID();
      const doc = new jsPDF('p', 'mm', 'a4');
      const pageWidth = doc.internal.pageSize.getWidth();
      const pageHeight = doc.internal.pageSize.getHeight();
      const margin = 15;
      let yPos = 20;

      // --- TÍTULO ---
      doc.setFontSize(16);
      doc.setFont('helvetica', 'bold');
      doc.text('REPORTE DE INSPECCIÓN VEHICULAR', pageWidth / 2, yPos, { align: 'center' });
      yPos += 15;

      // --- DATOS GENERALES ---
      doc.setFontSize(10);
      doc.setFont('helvetica', 'normal');
      
      // Fecha
      doc.text(`Fecha de Emisión: ${datos.hora || new Date().toLocaleString()}`, margin, yPos);
      yPos += 6;
      
      // Dirección (Manejo de texto largo)
      const direccionTexto = `Ubicación: ${datos.direccion || 'No registrada'}`;
      const splitDireccion = doc.splitTextToSize(direccionTexto, pageWidth - (margin * 2));
      doc.text(splitDireccion, margin, yPos);
      yPos += (splitDireccion.length * 5) + 2; 

      // Coordenadas (LATITUD Y LONGITUD) - ¡Aquí está lo que pediste!
      if (datos.coordenadas) {
        doc.setFontSize(9);
        // doc.setTextColor(100); // Gris oscurito para coordenadas
        // doc.text(`Coordenadas GPS: Lat ${datos.coordenadas.latitude}, Lon ${datos.coordenadas.longitude}`, margin, yPos);
        doc.setTextColor(0); // Regresamos a negro
        doc.setFontSize(10);
        yPos += 8;
      } else {
        yPos += 5;
      }

      // ID (Folio)
      doc.setFont('helvetica', 'bold');
      doc.text(`Folio ID: ${ID_GENERADO}`, margin, yPos);
      yPos += 15;

      // ==========================================
      //    TABLA (Con ID en encabezado)
      // ==========================================
      
      if (datos.checklist) {
        const colAnchoActividad = 90;
        const colAnchoOpcion = 25;
        const alturaFila = 8;
        
        const xActividad = margin;
        const xSi = margin + colAnchoActividad;
        const xNo = xSi + colAnchoOpcion;
        const xNa = xNo + colAnchoOpcion;

        // ENCABEZADO GRIS
        doc.setFillColor(220, 220, 220);
        
        doc.rect(xActividad, yPos, colAnchoActividad, alturaFila * 2, 'FD');
        doc.setFont('helvetica', 'bold');
        doc.setFontSize(11);
        doc.text("ACTIVIDAD", xActividad + 5, yPos + 10);

        // Encabezado ID
        doc.rect(xSi, yPos, colAnchoOpcion * 3, alturaFila, 'FD');
        doc.setFontSize(9);
        doc.text(`ID: ${ID_GENERADO}`, xSi + (colAnchoOpcion * 1.5), yPos + 5.5, { align: 'center' });

        // Sub-encabezados
        const ySub = yPos + alturaFila;
        doc.rect(xSi, ySub, colAnchoOpcion, alturaFila, 'FD');
        doc.rect(xNo, ySub, colAnchoOpcion, alturaFila, 'FD');
        doc.rect(xNa, ySub, colAnchoOpcion, alturaFila, 'FD');
        
        doc.setFontSize(9);
        doc.text("SI", xSi + (colAnchoOpcion/2), ySub + 5.5, { align: 'center' });
        doc.text("NO", xNo + (colAnchoOpcion/2), ySub + 5.5, { align: 'center' });
        doc.text("N/A", xNa + (colAnchoOpcion/2), ySub + 5.5, { align: 'center' });

        yPos += (alturaFila * 2);

        // CUERPO
        doc.setFont('helvetica', 'bold');
        doc.setFontSize(10);
        doc.setFillColor(240, 240, 240);
        doc.rect(xActividad, yPos, colAnchoActividad + (colAnchoOpcion * 3), alturaFila, 'FD');
        doc.text("LUCES", xActividad + 2, yPos + 5.5);
        yPos += alturaFila;

        doc.setFont('helvetica', 'normal');
        
        Object.entries(datos.checklist).forEach(([item, respuesta]) => {
            if (yPos > 260) { doc.addPage(); yPos = 20; }

            doc.rect(xActividad, yPos, colAnchoActividad, alturaFila);
            doc.text(item, xActividad + 2, yPos + 5.5);

            doc.rect(xSi, yPos, colAnchoOpcion, alturaFila);
            doc.rect(xNo, yPos, colAnchoOpcion, alturaFila);
            doc.rect(xNa, yPos, colAnchoOpcion, alturaFila);

            if (respuesta) {
                let xMark = 0;
                let color = [200, 200, 200];

                if (respuesta === "SI") { xMark = xSi; color = [76, 175, 80]; }
                else if (respuesta === "NO") { xMark = xNo; color = [244, 67, 54]; }
                else if (respuesta === "NA") { xMark = xNa; color = [255, 152, 0]; }

                doc.setFillColor(...color);
                doc.rect(xMark + 0.5, yPos + 0.5, colAnchoOpcion - 1, alturaFila - 1, 'F');
                doc.setTextColor(255, 255, 255);
                doc.setFont('helvetica', 'bold');
                doc.text(respuesta, xMark + (colAnchoOpcion/2), yPos + 5.5, { align: 'center' });
                doc.setTextColor(0, 0, 0);
                doc.setFont('helvetica', 'normal');
            }
            yPos += alturaFila;
        });
        yPos += 10;
      }

      // --- IMAGEN ---
      if (datos.imagen) {
        if (yPos + 80 > 270) { doc.addPage(); yPos = 20; }
        doc.setFont('helvetica', 'bold');
        doc.text("EVIDENCIA FOTOGRÁFICA:", margin, yPos);
        yPos += 5;
        doc.addImage(datos.imagen, 'JPEG', margin, yPos, 100, 75);
        yPos += 80;
      }

      // --- FIRMA ---
      if (datos.firma) {
        if (yPos + 40 > 270) { doc.addPage(); yPos = 20; }
        doc.setFont('helvetica', 'bold');
        doc.text("FIRMA DE CONFORMIDAD:", margin, yPos);
        yPos += 5;
        doc.addImage(datos.firma, 'PNG', margin, yPos, 60, 30);
        yPos += 35;
        doc.line(margin, yPos, margin + 60, yPos);
      }

      // --- PIE DE PÁGINA ---
      const totalPages = doc.internal.getNumberOfPages();
      for (let i = 1; i <= totalPages; i++) {
        doc.setPage(i);
        doc.setFontSize(8);
        doc.setTextColor(128);
        doc.setFont('helvetica', 'normal');
        doc.text('Documento generado automáticamente por SeguridApp', pageWidth / 2, pageHeight - 10, { align: 'center' });
        doc.text(`ID: ${ID_GENERADO} - Pág ${i}/${totalPages}`, pageWidth / 2, pageHeight - 5, { align: 'center' });
      }

      doc.save(`Reporte_${ID_GENERADO}.pdf`);

    } catch (error) {
      console.error('Error generando PDF:', error);
      alert('Error al generar PDF');
    }
  };

  // --- INTERFAZ VISUAL (Estilo Tarjeta) ---
  return (
    <div style={{
      maxWidth: '600px',
      margin: '0 auto',
      background: 'white',
      padding: '30px',
      borderRadius: '12px',
      boxShadow: '0 4px 15px rgba(0,0,0,0.1)',
      textAlign: 'center'
    }}>
      <h2 style={{ fontSize: '24px', fontWeight: 'bold', color: '#333', marginBottom: '10px' }}>
        Reporte Listo
      </h2>
      
      <p style={{ color: '#666', marginBottom: '30px' }}>
        Todos los datos han sido capturados correctamente.
      </p>

      <div style={{ 
        background: '#f8f9fa', 
        padding: '15px', 
        borderRadius: '8px', 
        marginBottom: '25px',
        textAlign: 'left',
        fontSize: '14px',
        borderLeft: '4px solid #4CAF50'
      }}>
        <p style={{ margin: '5px 0' }}><strong>Ubicación:</strong> {datos.direccion || 'Pendiente'}</p>
        {datos.coordenadas && (
           <p style={{ margin: '5px 0', color: '#666', fontSize: '12px' }}>
             📍 {datos.coordenadas.latitude}, {datos.coordenadas.longitude}
           </p>
        )}
        <p style={{ margin: '5px 0' }}><strong>Items verificados:</strong> {Object.keys(datos.checklist || {}).length}</p>
      </div>
      
      <button
        onClick={generarPDF}
        style={{
            background: '#4CAF50',
            color: 'white',
            padding: '16px 32px',
            fontSize: '16px',
            fontWeight: '600',
            border: 'none',
            borderRadius: '8px',
            cursor: 'pointer',
            boxShadow: '0 4px 6px rgba(76, 175, 80, 0.3)',
            transition: 'transform 0.1s',
            display: 'inline-flex',
            alignItems: 'center',
            gap: '10px'
        }}
        onMouseDown={(e) => e.target.style.transform = 'scale(0.98)'}
        onMouseUp={(e) => e.target.style.transform = 'scale(1)'}
      >
        <span>📄</span> Descargar PDF Oficial
      </button>
    </div>
  );
};

export default Reporte;


import React, { useRef, useState } from 'react';

const Ubicacion = ({ onUbicacionObtenida }) => {
  const [cargando, setCargando] = useState(false);
  const [error, setError] = useState(null);

  const obtenerUbicacion = () => {
    setCargando(true);
    setError(null);

    if (!navigator.geolocation) {
      setError('Geolocalización no soportada por el navegador');
      setCargando(false);
      return;
    }

    navigator.geolocation.getCurrentPosition(
      async (position) => {
        const { latitude, longitude } = position.coords;
        const coordenadas = { lat: latitude, lng: longitude };
        
        try {
          // Obtener dirección de OpenStreetMap
          const response = await fetch(
            `https://nominatim.openstreetmap.org/reverse?format=json&lat=${latitude}&lon=${longitude}&zoom=18&addressdetails=1`,
            {
              headers: {
                'Accept-Language': 'es',
                'User-Agent': 'SeguriApp/1.0'
              }
            }
          );
          
          const data = await response.json();
          const direccion = data.display_name || 'Dirección no disponible';
          
          onUbicacionObtenida(
            `Lat: ${latitude.toFixed(6)}, Lon: ${longitude.toFixed(6)}`,
            direccion,
            coordenadas
          );
        } catch (err) {
          console.error('Error obteniendo dirección:', err);
          onUbicacionObtenida(
            `Lat: ${latitude.toFixed(6)}, Lon: ${longitude.toFixed(6)}`,
            'Error al obtener dirección detallada',
            coordenadas
          );
        }
        
        setCargando(false);
      },
      (err) => {
        console.error('Error de geolocalización:', err);
        setError('No se pudo obtener la ubicación. Verifica los permisos.');
        setCargando(false);
      },
      {
        enableHighAccuracy: true,
        timeout: 10000,
        maximumAge: 0
      }
    );
  };

  return (
    <div className="bg-white p-6 rounded-lg shadow-lg">
      <h2 className="text-xl font-bold mb-4">2. Registrar Ubicación</h2>
      
      <div className="mb-6">
        <p className="text-gray-600 mb-4">
          Para registrar la ubicación exacta del incidente, necesitamos tu permiso.
        </p>
        
        <button
          onClick={obtenerUbicacion}
          disabled={cargando}
          className={`w-full py-3 rounded-lg font-medium ${
            cargando 
              ? 'bg-gray-400 cursor-not-allowed' 
              : 'bg-green-600 hover:bg-green-700 text-white'
          }`}
        >
          {cargando ? (
            <span className="flex items-center justify-center">
              <svg className="animate-spin h-5 w-5 mr-3 text-white" viewBox="0 0 24 24">
                <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" fill="none" />
                <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z" />
              </svg>
              Obteniendo ubicación...
            </span>
          ) : (
            '📍 Obtener Ubicación Actual'
          )}
        </button>
      </div>

      {error && (
        <div className="bg-red-50 border border-red-200 text-red-700 px-4 py-3 rounded-lg mb-4">
          <strong>Error:</strong> {error}
        </div>
      )}

      <div className="bg-blue-50 border border-blue-200 p-4 rounded-lg">
        <h3 className="font-bold text-blue-800 mb-2">¿Por qué necesitamos tu ubicación?</h3>
        <ul className="text-blue-700 text-sm space-y-1">
          <li>• Registro exacto del lugar del incidente</li>
          <li>• Cumplimiento de requisitos legales</li>
          <li>• Posible envío de asistencia</li>
          <li>• Evidencia georreferenciada</li>
        </ul>
      </div>
    </div>
  );
};

export default Ubicacion;

    imagen: null,
    ubicacion: null,
    direccion: '',
    hora: '',
    firma: null,
    coordenadas: null
  });

  const actualizarDatos = (nuevos) => {
    setDatos(prev => ({ ...prev, ...nuevos }));
  };

  // --- LÓGICA DE VALIDACIÓN ESTRICTA ---
  const siguientePaso = () => {
    
    // -------------------------------------------------
    // 1. VALIDACIÓN DEL CHECKLIST
    // -------------------------------------------------
    if (paso === 1) {
      const itemsTotales = Object.values(estructura).flat();
      const faltantes = itemsTotales.filter(item => !checklist[item]);

      // Regla 1: Todo debe estar contestado
      if (faltantes.length > 0) {
        alert(`⚠️ NO PUEDES AVANZAR.\n\nFaltan por contestar:\n- ${faltantes.join("\n- ")}`);
        return; 
      }

      // Regla 2: Decidir si la foto es obligatoria o se salta
      const hayObservaciones = Object.values(checklist).some(val => val === "NO" || val === "NA");

      if (!hayObservaciones) {
        // Todo es "SI" -> Saltamos la foto, vamos directo a Ubicación (Paso 3)
        actualizarDatos({ imagen: null }); 
        setPaso(3); 
        return;
      }
      // Si hay fallos, avanzamos normal al Paso 2 (Foto)
    }

    // -------------------------------------------------
    // 2. VALIDACIÓN DE FOTO (OBLIGATORIA SI HAY FALLOS)
    // -------------------------------------------------
    if (paso === 2) {
      if (!datos.imagen) {
        alert("⚠️ FOTO OBLIGATORIA.\n\nComo reportaste fallos (NO/NA), debes tomar la evidencia fotográfica antes de continuar.");
        return;
      }
    }

    // -------------------------------------------------
    // 3. VALIDACIÓN DE UBICACIÓN (OBLIGATORIA)
    // -------------------------------------------------
    if (paso === 3) {
      if (!datos.ubicacion) {
        alert("⚠️ UBICACIÓN REQUERIDA.\n\nEs necesario registrar la ubicación GPS para validar el reporte.");
        return;
      }
    }

    // -------------------------------------------------
    // 4. VALIDACIÓN DE FIRMA (OBLIGATORIA)
    // -------------------------------------------------
    if (paso === 4) {
      if (!datos.firma) {
        alert("⚠️ FIRMA FALTANTE.\n\nDebes firmar y presionar el botón 'Guardar Firma' para continuar.");
        return;
      }
    }

    // Si pasamos todas las barreras, avanzamos
    if (paso < 5) setPaso(p => p + 1);
  };

  const anteriorPaso = () => {
    // Solo permitimos regresar del Paso 2 al 1 para editar el checklist
    if (paso === 2) {
      setPaso(1);
    }
    // Si estás en el paso 3 (y te saltaste el 2), regresarías al 1
    else if (paso === 3) {
       // Verificamos si veníamos de salto o no
       const hayObservaciones = Object.values(checklist).some(val => val === "NO" || val === "NA");
       if (!hayObservaciones) {
         setPaso(1);
       } else {
         // Si había observaciones, el paso anterior lógico es el 2
         setPaso(2);
       }
    }
  };

  const guardarInspeccion = () => {
    const historial = JSON.parse(localStorage.getItem("inspecciones")) || [];
    const nueva = {
      id: crypto.randomUUID(),
      fecha: new Date().toISOString(),
      checklist,
      ...datos
    };
    historial.push(nueva);
    localStorage.setItem("inspecciones", JSON.stringify(historial));
  };

  return (
    <div className="app">
      <header>
        <h1>SeguridApp</h1>
        <p className="subtitulo">Registro de Inspección</p>

        {/* Barra de Progreso */}
        <div className="progreso-container">
          <div className="progreso">
            <div className={`paso ${paso >= 1 ? 'activo' : ''}`}>1. Lista</div>
            <div className={`paso ${paso >= 2 ? 'activo' : ''} ${paso === 3 && !datos.imagen && paso > 1 ? 'saltado' : ''}`}>
               2. Foto
            </div>
            <div className={`paso ${paso >= 3 ? 'activo' : ''}`}>3. Ubic.</div>
            <div className={`paso ${paso >= 4 ? 'activo' : ''}`}>4. Firma</div>
            <div className={`paso ${paso >= 5 ? 'activo' : ''}`}>5. Fin</div>
          </div>
        </div>
      </header>

      <main className="contenido">
        {paso === 1 && (
          <Checklist onChange={setChecklist} datosPrevios={checklist} />
        )}

        {paso === 2 && (
          <div className="fade-in">
            <div className="alerta-foto">
              📸 <strong>Evidencia Requerida:</strong><br/>
              Detectamos items marcados como "NO" o "N/A". La foto es obligatoria.
            </div>
            <Camara 
              onCapture={(imagen, hora) => {
                actualizarDatos({ imagen, hora });
                // Aquí quitamos el avance automático para que el usuario 
                // vea que ya se tomó la foto y le de él mismo a "Siguiente"
              }}
            />
          </div>
        )}

        {paso === 3 && (
          <Ubicacion 
            onUbicacionObtenida={(ubicacion, direccion, coordenadas) => {
              actualizarDatos({ ubicacion, direccion, coordenadas });
            }}
          />
        )}

        {paso === 4 && (
          <Firma 
            onFirmaCompletada={(firma) => {
              actualizarDatos({ firma });
            }}
          />
        )}

        {paso === 5 && (
          <Reporte 
            datos={{ ...datos, checklist }} 
            onGuardar={guardarInspeccion}
          />
        )}

        {/* Botones de Navegación */}
        <div className="navegacion">
          {/* BOTÓN ANTERIOR: Solo visible en paso 2 (o paso 3 si saltó) */}
          {(paso === 2 || (paso === 3 && !datos.imagen)) && (
            <button onClick={anteriorPaso} className="btn btn-secundario">
              ← Corregir Checklist
            </button>
          )}

          {/* Espaciador por si no hay botón anterior */}
          {paso !== 2 && !(paso === 3 && !datos.imagen) && <div></div>}

          {/* BOTÓN SIGUIENTE */}
          {paso < 5 && (
            <button 
              onClick={siguientePaso} 
              className={`btn btn-principal ${
                // Estilos condicionales para dar feedback visual (opcional)
                (paso === 2 && !datos.imagen) || 
                (paso === 3 && !datos.ubicacion) || 
                (paso === 4 && !datos.firma) 
                  ? 'btn-deshabilitado-visual' : ''
              }`}
            >
              {paso === 4 ? 'Finalizar y Ver Reporte →' : 'Siguiente →'}
            </button>
          )}
        </div>
      </main>

      <style>{`
        /* Reset básico */
        * { box-sizing: border-box; }
        
        .app {
          padding: 15px;
          max-width: 800px;
          margin: 0 auto;
          font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif;
          min-height: 100vh;
          background-color: #f5f7fa;
        }

        header { text-align: center; margin-bottom: 20px; }
        h1 { margin: 0; color: #1a365d; font-size: 1.8rem; }
        .subtitulo { margin: 5px 0 15px 0; color: #718096; font-size: 0.9rem; }

        .progreso-container { overflow-x: auto; }
        .progreso {
          display: flex; gap: 4px; justify-content: space-between;
          background: #e2e8f0; padding: 4px; border-radius: 8px;
        }
        .paso {
          flex: 1; padding: 8px 4px; text-align: center; border-radius: 6px;
          font-size: 0.75rem; color: #718096; background: white;
          white-space: nowrap; overflow: hidden; text-overflow: ellipsis;
        }
        .paso.activo { background: #2196F3; color: white; font-weight: bold; }
        .paso.saltado { background: #90CDF4; color: white; opacity: 0.6; }

        .contenido {
          background: white; padding: 20px; border-radius: 12px;
          box-shadow: 0 4px 6px rgba(0,0,0,0.05); margin-bottom: 20px;
        }

        .alerta-foto {
          background-color: #fff3cd; color: #856404; padding: 15px;
          border-radius: 6px; margin-bottom: 20px; border-left: 5px solid #ffeeba;
        }

        .navegacion {
          display: flex; justify-content: space-between; gap: 15px; margin-top: 25px;
        }

        .btn {
          padding: 14px 20px; border: none; border-radius: 8px; cursor: pointer;
          font-size: 1rem; font-weight: 600; transition: all 0.2s; flex: 1;
        }
        .btn-principal { background: #2196F3; color: white; box-shadow: 0 4px 6px rgba(33, 150, 243, 0.3); }
        .btn-principal:active { transform: scale(0.98); }
        .btn-secundario { background: #CBD5E0; color: #4A5568; }
        
        /* Estilo visual para indicar que falta completar algo (opcional, solo visual) */
        .btn-deshabilitado-visual { opacity: 0.7; }

        .fade-in { animation: fadeIn 0.4s ease-in; }
        @keyframes fadeIn { from { opacity: 0; } to { opacity: 1; } }

        @media (max-width: 600px) {
          .navegacion { flex-direction: column-reverse; }
          .btn { width: 100%; }
        }
      `}</style>
    </div>
  );
}

export default App;
