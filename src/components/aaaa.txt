import React, { useRef, useState } from 'react';
import Webcam from 'react-webcam';

const Camara = ({ onCapture }) => {
  const webcamRef = useRef(null);
  const [imgSrc, setImgSrc] = useState(null);

  const capture = () => {
    const imageSrc = webcamRef.current.getScreenshot();
    setImgSrc(imageSrc);
    const hora = new Date().toLocaleString('es-ES', {
      dateStyle: 'full',
      timeStyle: 'medium'
    });
    onCapture(imageSrc, hora);
  };

  return (
    <div style={{ background: 'white', padding: '20px', borderRadius: '12px', boxShadow: '0 4px 6px rgba(0,0,0,0.1)' }}>
      <h2 style={{ fontSize: '1.25rem', fontWeight: 'bold', marginBottom: '15px' }}>
        Evidencia Fotográfica
      </h2>
      
      {/* Forzamos a que sea en columna (vertical) para que no rompa en celulares */}
      <div style={{ display: 'flex', flexDirection: 'column', gap: '20px' }}>
        
        {/* CONTENEDOR DE LA CÁMARA (Con límite de altura para que no empuje los botones) */}
        <div style={{ flex: 1 }}>
          <div style={{ 
            border: '2px solid #ccc', 
            borderRadius: '8px', 
            overflow: 'hidden',
            backgroundColor: '#000',
            display: 'flex',
            justifyContent: 'center',
            maxHeight: '50vh' /* Esto evita que la cámara sea gigante en el celular */
          }}>
            {imgSrc ? (
              <img src={imgSrc} alt="Captura" style={{ width: '100%', height: 'auto', maxHeight: '50vh', objectFit: 'contain' }} />
            ) : (
              <Webcam
                audio={false}
                ref={webcamRef}
                screenshotFormat="image/jpeg"
                style={{ width: '100%', height: 'auto', maxHeight: '50vh', objectFit: 'cover' }}
                videoConstraints={{
                  facingMode: 'environment'
                }}
              />
            )}
          </div>
          
          <div style={{ display: 'flex', gap: '10px', marginTop: '15px' }}>
            <button
              onClick={capture}
              style={{
                flex: 1,
                background: '#2196F3',
                color: 'white',
                padding: '15px',
                borderRadius: '8px',
                border: 'none',
                fontWeight: 'bold',
                fontSize: '1rem',
                cursor: 'pointer'
              }}
            >
              {imgSrc ? 'Tomar Otra Foto' : 'Capturar Foto'}
            </button>
            
            {imgSrc && (
              <button
                onClick={() => setImgSrc(null)}
                style={{
                  flex: 1,
                  background: '#757575',
                  color: 'white',
                  padding: '15px',
                  borderRadius: '8px',
                  border: 'none',
                  fontWeight: 'bold',
                  fontSize: '1rem',
                  cursor: 'pointer'
                }}
              >
                Reintentar
              </button>
            )}
          </div>
        </div>
        
        {/* Instrucciones debajo */}
        <div style={{ background: '#f8f9fa', padding: '15px', borderRadius: '8px' }}>
          <h3 style={{ fontWeight: 'bold', margin: '0 0 10px 0', fontSize: '1rem' }}>Instrucciones:</h3>
          <ul style={{ paddingLeft: '20px', margin: 0, fontSize: '0.9rem', color: '#555', lineHeight: '1.5' }}>
            <li>Asegúrese de que el vehículo esté completamente visible</li>
            <li>Capture la placa si es posible</li>
            <li>Mantenga el dispositivo estable</li>
          </ul>
        </div>
        
      </div>
    </div>
  );
};

export default Camara;

export default function Checklist({ onChange, datosPrevios }) {
  const [respuestas, setRespuestas] = useState(datosPrevios || {});
  
  // NUEVO: Estado para saber en qué sección estamos (empezamos en la 0)
  const [seccionActiva, setSeccionActiva] = useState(0);
  
  // Sacamos los nombres de las secciones ["Luces", "Niveles", ...]
  const nombresSecciones = Object.keys(estructura);
  const seccionActual = nombresSecciones[seccionActiva];
  const itemsActuales = estructura[seccionActual];

  const marcar = (item, valor) => {
    const nuevo = { ...respuestas, [item]: valor };
    setRespuestas(nuevo);
    onChange(nuevo);
  };

  const obtenerColor = (item, op) => {
    if (respuestas[item] === op) {
      if (op === "SI") return "#4CAF50"; // Verde
      if (op === "NO") return "#F44336"; // Rojo
      if (op === "NA") return "#FF9800"; // Naranja
    }
    return "#e0e0e0"; // Gris
  };

  // Funciones de navegación interna del checklist
  const siguienteSeccion = () => {
    if (seccionActiva < nombresSecciones.length - 1) {
      setSeccionActiva(seccionActiva + 1);
    }
  };

  const anteriorSeccion = () => {
    if (seccionActiva > 0) {
      setSeccionActiva(seccionActiva - 1);
    }
  };

  return (
    <div style={{ maxWidth: '600px', margin: '0 auto' }}>
      <h2 style={{ textAlign: 'center', marginBottom: 15, fontSize: '1.4rem' }}>Inspección de Unidad</h2>
      
      {/* Pestañas indicadoras (Mini menú arriba) */}
      <div style={{ display: 'flex', overflowX: 'auto', gap: '8px', marginBottom: '20px', paddingBottom: '5px' }}>
        {nombresSecciones.map((sec, index) => {
          // Checamos si la sección ya está completamente contestada
          const itemsDeEstaSeccion = estructura[sec];
          const contestados = itemsDeEstaSeccion.filter(i => respuestas[i]).length;
          const completa = contestados === itemsDeEstaSeccion.length;

          return (
            <button
              key={sec}
              onClick={() => setSeccionActiva(index)}
              style={{
                flex: '0 0 auto',
                padding: '8px 12px',
                background: seccionActiva === index ? '#2196F3' : (completa ? '#E8F5E9' : '#f0f0f0'),
                color: seccionActiva === index ? 'white' : (completa ? '#2E7D32' : '#333'),
                border: completa && seccionActiva !== index ? '1px solid #A5D6A7' : 'none',
                borderRadius: '20px',
                fontSize: '0.85rem',
                fontWeight: seccionActiva === index ? 'bold' : 'normal',
                cursor: 'pointer',
                whiteSpace: 'nowrap'
              }}
            >
              {sec} {completa ? '✓' : ''}
            </button>
          );
        })}
      </div>

      {/* Contenido de la sección activa */}
      <div style={{ background: '#fff', padding: '15px', borderRadius: '8px', boxShadow: '0 2px 8px rgba(0,0,0,0.05)' }}>
        <h3 style={{ margin: '0 0 15px 0', borderBottom: '2px solid #2196F3', paddingBottom: '8px', color: '#1a365d' }}>
          {seccionActual}
        </h3>

        {itemsActuales.map(item => (
          <div key={item} style={{ 
            display: "flex", 
            justifyContent: "space-between", 
            alignItems: "center",
            marginBottom: '12px',
            padding: "8px 0",
            borderBottom: "1px solid #f0f0f0"
          }}>
            <span style={{ flex: 1, fontWeight: '500', fontSize: '0.9rem', paddingRight: '10px' }}>{item}</span>
            <div style={{ display: "flex", gap: '6px' }}>
              {["SI", "NO", "NA"].map(op => (
                <button
                  key={op}
                  onClick={() => marcar(item, op)}
                  style={{
                    padding: "8px 14px",
                    background: obtenerColor(item, op),
                    color: respuestas[item] === op ? "white" : "#333",
                    border: respuestas[item] === op ? "none" : "1px solid #ccc",
                    borderRadius: '6px',
                    cursor: "pointer",
                    fontWeight: "bold",
                    fontSize: "0.85rem",
                    transition: "all 0.1s"
                  }}
                >
                  {op}
                </button>
              ))}
            </div>
          </div>
        ))}
      </div>

      {/* Navegación entre secciones dentro del Checklist */}
      <div style={{ display: 'flex', justifyContent: 'space-between', marginTop: '20px' }}>
        <button 
          onClick={anteriorSeccion}
          disabled={seccionActiva === 0}
          style={{
            padding: '10px 15px', background: seccionActiva === 0 ? '#e0e0e0' : '#757575', color: 'white', border: 'none', borderRadius: '6px', cursor: seccionActiva === 0 ? 'not-allowed' : 'pointer'
          }}
        >
          ← Anterior
        </button>

        {seccionActiva < nombresSecciones.length - 1 ? (
          <button 
            onClick={siguienteSeccion}
            style={{ padding: '10px 15px', background: '#2196F3', color: 'white', border: 'none', borderRadius: '6px', cursor: 'pointer' }}
          >
            Siguiente Sección →
          </button>
        ) : (
          <div style={{ padding: '10px 15px', color: '#4CAF50', fontWeight: 'bold' }}>
            Lista Terminada ✓
          </div>
        )}
      </div>
      
    </div>
  );
}

import React, { useState } from 'react';
import { PDFDocument } from 'pdf-lib';

export default function DatosGenerales({ datos, onChange, onCargarDatos }) {
  const [cargandoPDF, setCargandoPDF] = useState(false);

  const handleChange = (e) => {
    const { name, value } = e.target;
    onChange({ [name]: value });
  };

  // 👇 LÓGICA PARA LEER EL PDF Y EXTRAER LOS METADATOS 👇
  const handleSubirPDF = async (e) => {
    const file = e.target.files[0];
    if (!file) return;

    setCargandoPDF(true);
    try {
      const arrayBuffer = await file.arrayBuffer();
      const pdfDoc = await PDFDocument.load(arrayBuffer);
      
      // Extraemos la información oculta que guardamos en "keywords"
      const metadatosOcultos = pdfDoc.getKeywords(); 

      if (metadatosOcultos) {
        const datosExtraidos = JSON.parse(metadatosOcultos);
        onCargarDatos(datosExtraidos); // Enviamos los datos a la App
        alert("✅ ¡Reporte anterior cargado con éxito! Se han rellenado tus datos y el checklist.");
      } else {
        alert("⚠️ Este PDF no contiene datos de SeguridApp o es un formato antiguo.");
      }
    } catch (error) {
      console.error(error);
      alert("❌ Hubo un error al leer el documento PDF.");
    }
    setCargandoPDF(false);
    e.target.value = null; // Resetea el input
  };

  const inputStyle = {
    width: '100%', padding: '10px', marginTop: '5px', marginBottom: '15px',
    border: '1px solid #ccc', borderRadius: '6px', fontSize: '1rem'
  };
  const labelStyle = { fontWeight: 'bold', color: '#333', fontSize: '0.9rem' };

  return (
    <div style={{ background: '#fff', padding: '20px', borderRadius: '8px', boxShadow: '0 2px 8px rgba(0,0,0,0.05)' }}>
      
      {/* SECCIÓN DE CARGA MAGICA */}
      <div style={{ background: '#E3F2FD', padding: '15px', borderRadius: '8px', marginBottom: '25px', textAlign: 'center', border: '2px dashed #2196F3' }}>
        <h3 style={{ margin: '0 0 10px 0', color: '#1565C0' }}>¿Continuar inspección de la semana pasada?</h3>
        <p style={{ fontSize: '0.85rem', color: '#555', marginBottom: '15px' }}>
          Sube el PDF de la semana anterior y el sistema rellenará los datos automáticamente.
        </p>
        <label style={{ background: '#2196F3', color: 'white', padding: '10px 20px', borderRadius: '6px', cursor: 'pointer', fontWeight: 'bold', display: 'inline-block' }}>
          {cargandoPDF ? 'Leyendo documento...' : '📄 Subir PDF Anterior'}
          <input type="file" accept="application/pdf" onChange={handleSubirPDF} style={{ display: 'none' }} disabled={cargandoPDF} />
        </label>
      </div>

      <h2 style={{ textAlign: 'center', marginBottom: '20px', color: '#1a365d' }}>Datos del Reporte Actual</h2>
      
      <div style={{ display: 'flex', gap: '15px', flexWrap: 'wrap' }}>
        <div style={{ flex: '1 1 45%' }}>
          <label style={labelStyle}>NO. ECO (Económico):</label>
          <input type="text" name="noEco" value={datos.noEco || ''} onChange={handleChange} style={inputStyle} placeholder="Ej. 12345" />
        </div>
        <div style={{ flex: '1 1 45%' }}>
          <label style={labelStyle}>Área o Depto:</label>
          <input type="text" name="area" value={datos.area || ''} onChange={handleChange} style={inputStyle} placeholder="Ej. Distribución" />
        </div>
      </div>

      <div style={{ display: 'flex', gap: '15px', flexWrap: 'wrap' }}>
        <div style={{ flex: '1 1 100%' }}>
          <label style={labelStyle}>Periodo de Revisión:</label>
          <input type="text" name="periodo" value={datos.periodo || ''} onChange={handleChange} style={inputStyle} placeholder="Ej. Semana del 12 al 16" />
        </div>
      </div>

      <div style={{ display: 'flex', gap: '15px', flexWrap: 'wrap' }}>
        <div style={{ flex: '1 1 30%' }}>
          <label style={labelStyle}>Mes:</label>
          <input type="text" name="mes" value={datos.mes || ''} onChange={handleChange} style={inputStyle} placeholder="Ej. Febrero" />
        </div>
        <div style={{ flex: '1 1 30%' }}>
          <label style={labelStyle}>Año:</label>
          <input type="number" name="anio" value={datos.anio || ''} onChange={handleChange} style={inputStyle} placeholder="Ej. 2026" />
        </div>
        <div style={{ flex: '1 1 30%' }}>
          <label style={labelStyle}>Kilometraje:</label>
          <input type="number" name="kilometraje" value={datos.kilometraje || ''} onChange={handleChange} style={inputStyle} placeholder="Ej. 85000" />
        </div>
      </div>

      <hr style={{ margin: '20px 0', border: 'none', borderTop: '1px solid #eee' }} />
      <h3 style={{ marginBottom: '15px', color: '#1a365d' }}>Datos del Trabajador</h3>

      <div style={{ display: 'flex', gap: '15px', flexWrap: 'wrap' }}>
        <div style={{ flex: '1 1 60%' }}>
          <label style={labelStyle}>Nombre Completo:</label>
          <input type="text" name="nombre" value={datos.nombre || ''} onChange={handleChange} style={inputStyle} placeholder="Tu nombre" />
        </div>
        <div style={{ flex: '1 1 35%' }}>

          <input type="text" name="rpe" value={datos.rpe || ''} onChange={handleChange} style={inputStyle} placeholder="Tu RPE" />
        </div>
      </div>
    </div>
  );
}




import React, { useRef, useState } from 'react';
import SignatureCanvas from 'react-signature-canvas';

const Firma = ({ onFirmaCompletada }) => {
  const firmaRef = useRef(null);
  const [firmaData, setFirmaData] = useState(null);

  const limpiarFirma = () => {
    firmaRef.current.clear();
    setFirmaData(null);
  };
const guardarFirma = () => {
  if (!firmaRef.current || firmaRef.current.isEmpty()) {
    alert('Por favor, firma en el espacio proporcionado');
    return;
  }

  const firmaBase64 = firmaRef.current
    .getCanvas()
    .toDataURL('image/png');

  setFirmaData(firmaBase64);
  onFirmaCompletada(firmaBase64);
};


  return (
    <div className="bg-white p-6 rounded-lg shadow-lg">
      <h2 className="text-xl font-bold mb-4">3. Firma Digital de Confirmación</h2>
      
      <div className="mb-4">
        <p className="text-gray-600 mb-2">
          Firma en el recuadro para confirmar la veracidad de la información proporcionada.
        </p>
      </div>

      <div className="border-2 border-gray-300 rounded-lg p-2 bg-white mb-4">
        <SignatureCanvas
          ref={firmaRef}
          canvasProps={{
            width: 600,
            height: 200,
            className: 'w-full h-48 border border-gray-200 rounded'
          }}
          penColor="black"
          backgroundColor="white"
        />
      </div>

      <div className="flex gap-4 mb-6">
        <button
          onClick={limpiarFirma}
          className="flex-1 bg-gray-200 text-gray-800 py-2 rounded-lg hover:bg-gray-300"
        >
          Borrar Firma
        </button>
        
        <button
          onClick={guardarFirma}
          className="flex-1 bg-blue-600 text-white py-2 rounded-lg hover:bg-blue-700"
        >
          Guardar Firma
        </button>
      </div>

      {firmaData && (
        <div className="border border-green-200 bg-green-50 p-4 rounded-lg">
          <div className="flex items-center text-green-700">
            <svg className="w-5 h-5 mr-2" fill="currentColor" viewBox="0 0 20 20">
              <path fillRule="evenodd" d="M10 18a8 8 0 100-16 8 8 0 000 16zm3.707-9.293a1 1 0 00-1.414-1.414L9 10.586 7.707 9.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414 0l4-4z" clipRule="evenodd" />
            </svg>
            <span>Firma guardada correctamente</span>
          </div>
        </div>
      )}

      <div className="mt-6 text-sm text-gray-500">
        <p><strong>Nota:</strong> Esta firma tiene validez legal como confirmación de los datos reportados.</p>
      </div>
    </div>
  );
};

export default Firma;




import React from 'react';
import { jsPDF } from 'jspdf';
import { estructura } from './Checklist';

const Reporte = ({ datos }) => {

  const generarID = () => {
    const textoUbicacion = (datos.direccion || "").toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
    let codigo = "SW000"; 
    if (textoUbicacion.includes("merida")) { codigo = "SW010"; }
    // ... (Puedes agregar el resto de las validaciones de ciudades aquí como las tenías)
    const now = new Date();
    return `${codigo}${now.getDate()}${now.getMonth()+1}${now.getFullYear()}${now.getHours()}${now.getMinutes()}`;
  };

  const generarPDF = async () => {
    try {
      const ID_GENERADO = generarID();
      const doc = new jsPDF('p', 'mm', 'a4');



      // 👇 --- AQUÍ GUARDAMOS LA MAGIA (LOS METADATOS) --- 👇
      const f = datos.form || {};
      const estadoParaGuardar = JSON.stringify({ 
        form: f, 
        checklist: datos.checklist 
      });

      doc.setProperties({
        title: `Verificacion_${f.noEco || 'Vehiculo'}_${ID_GENERADO}`,
        subject: 'Inspeccion Vehicular CFE',
        author: f.nombre || 'Trabajador CFE',
        keywords: estadoParaGuardar, // <--- AQUÍ SE GUARDA TODA LA INFO
        creator: 'SeguridApp'
      });
      // 👆 ------------------------------------------------



      const pageWidth = doc.internal.pageSize.getWidth();
      const pageHeight = doc.internal.pageSize.getHeight();
      const margin = 15;
      let yPos = 15;

      // const f = datos.form || {}; // Datos del Paso 1

      // 1. CÓDIGO FORMA SH-209 (Esquina Superior Derecha)
      doc.setFontSize(9);
      doc.setFont('helvetica', 'normal');
      doc.text("Forma,- SH-209", pageWidth - margin, yPos, { align: 'right' });
      yPos += 10;

      // 2. TÍTULOS OFICIALES Y ESPACIOS PARA LOGOS
      // Cuadros grises como "Placeholder" para los logos futuros
      doc.setFillColor(230, 230, 230);
      doc.rect(margin, yPos, 25, 25, 'F'); // Logo Izquierdo
      doc.rect(pageWidth - margin - 25, yPos, 25, 25, 'F'); // Logo Derecho
      doc.setFontSize(7);
      doc.text("LOGO CFE", margin + 12.5, yPos + 13, { align: 'center' });
      doc.text("LOGO SUTERM", pageWidth - margin - 12.5, yPos + 13, { align: 'center' });

      doc.setFontSize(13);
      doc.setFont('helvetica', 'bold');
      doc.text("COMISION FEDERAL DE ELECTRICIDAD", pageWidth / 2, yPos + 5, { align: 'center' });
      doc.setFontSize(11);
      doc.text("DIVISION DE DISTRIBUCION PENINSULAR", pageWidth / 2, yPos + 12, { align: 'center' });
      doc.setFontSize(9);
      doc.text("VERIFICACION SEMANAL DE VEHÍCULO DE TRABAJO, PERSONAL OFICINA", pageWidth / 2, yPos + 20, { align: 'center' });
      
      yPos += 35;

      // 3. DATOS GENERALES DEL VEHÍCULO
      doc.setFontSize(9);
      doc.setFont('helvetica', 'bold');
      
      // Fila 1
      doc.text(`NO. ECO:`, margin, yPos);
      doc.setFont('helvetica', 'normal');
      doc.text(`${f.noEco || '______'}`, margin + 18, yPos);
      
      doc.setFont('helvetica', 'bold');
      doc.text(`ÁREA O DEPTO:`, margin + 80, yPos);
      doc.setFont('helvetica', 'normal');
      doc.text(`${f.area || '______'}`, margin + 110, yPos);
      yPos += 7;

      // Fila 2
      doc.setFont('helvetica', 'bold');
      doc.text(`PERIODO DE REVISIÓN:`, margin, yPos);
      doc.setFont('helvetica', 'normal');
      doc.text(`${f.periodo || '______'}`, margin + 42, yPos);

      doc.setFont('helvetica', 'bold');
      doc.text(`MES:`, margin + 85, yPos);
      doc.setFont('helvetica', 'normal');
      doc.text(`${f.mes || '___'}`, margin + 95, yPos);

      doc.setFont('helvetica', 'bold');
      doc.text(`KILOMETRAJE:`, margin + 120, yPos);
      doc.setFont('helvetica', 'normal');
      doc.text(`${f.kilometraje || '___'}`, margin + 148, yPos);

      doc.setFont('helvetica', 'bold');
      doc.text(`AÑO:`, margin + 168, yPos);
      doc.setFont('helvetica', 'normal');
      doc.text(`${f.anio || '___'}`, margin + 178, yPos);
      
      yPos += 12;

      // 4. TABLA DEL CHECKLIST
      if (datos.checklist) {
        const colAnchoActividad = 90;
        const colAnchoOpcion = 25;
        const alturaFila = 8;
        const xActividad = margin;
        const xSi = margin + colAnchoActividad;
        const xNo = xSi + colAnchoOpcion;
        const xNa = xNo + colAnchoOpcion;

        doc.setFillColor(220, 220, 220);
        doc.rect(xActividad, yPos, colAnchoActividad, alturaFila * 2, 'FD');
        doc.setFont('helvetica', 'bold');
        doc.text("ACTIVIDAD", xActividad + 5, yPos + 10);

        doc.rect(xSi, yPos, colAnchoOpcion * 3, alturaFila, 'FD');
        doc.text(`ID: ${ID_GENERADO}`, xSi + (colAnchoOpcion * 1.5), yPos + 5.5, { align: 'center' });

        const ySub = yPos + alturaFila;
        doc.rect(xSi, ySub, colAnchoOpcion, alturaFila, 'FD');
        doc.rect(xNo, ySub, colAnchoOpcion, alturaFila, 'FD');
        doc.rect(xNa, ySub, colAnchoOpcion, alturaFila, 'FD');
        doc.text("SI", xSi + (colAnchoOpcion/2), ySub + 5.5, { align: 'center' });
        doc.text("NO", xNo + (colAnchoOpcion/2), ySub + 5.5, { align: 'center' });
        doc.text("N/A", xNa + (colAnchoOpcion/2), ySub + 5.5, { align: 'center' });

        yPos += (alturaFila * 2);

        Object.entries(estructura).forEach(([nombreSeccion, itemsSeccion]) => {
            if (yPos > 240) { doc.addPage(); yPos = 20; }
            doc.setFillColor(200, 200, 200);
            doc.rect(xActividad, yPos, colAnchoActividad + (colAnchoOpcion * 3), alturaFila, 'FD');
            doc.text(nombreSeccion.toUpperCase(), xActividad + 2, yPos + 5.5);
            yPos += alturaFila;

            doc.setFont('helvetica', 'normal');
            itemsSeccion.forEach((item) => {
                const respuesta = datos.checklist[item];
                if (respuesta) {
                    if (yPos > 240) { doc.addPage(); yPos = 20; }
                    doc.rect(xActividad, yPos, colAnchoActividad, alturaFila);
                    doc.text(item, xActividad + 2, yPos + 5.5);
                    doc.rect(xSi, yPos, colAnchoOpcion, alturaFila);
                    doc.rect(xNo, yPos, colAnchoOpcion, alturaFila);
                    doc.rect(xNa, yPos, colAnchoOpcion, alturaFila);

                    let xMark = 0, color = [200, 200, 200];
                    if (respuesta === "SI") { xMark = xSi; color = [76, 175, 80]; }
                    else if (respuesta === "NO") { xMark = xNo; color = [244, 67, 54]; }
                    else if (respuesta === "NA") { xMark = xNa; color = [255, 152, 0]; }

                    doc.setFillColor(...color);
                    doc.rect(xMark + 0.5, yPos + 0.5, colAnchoOpcion - 1, alturaFila - 1, 'F');
                    doc.setTextColor(255);
                    doc.setFont('helvetica', 'bold');
                    doc.text(respuesta, xMark + (colAnchoOpcion/2), yPos + 5.5, { align: 'center' });
                    doc.setTextColor(0);
                    doc.setFont('helvetica', 'normal');
                    yPos += alturaFila;
                }
            });
        });
        yPos += 10;
      }

      // 5. NOTA CHIQUITA EN LA PRIMERA PÁGINA (Letra tamaño 7)
      doc.setPage(1); // Nos aseguramos de escribir esto en la página 1
      doc.setFontSize(7);
      doc.setFont('helvetica', 'bold');
      const notaAviso = "NOTA: SI CUMPLISTE CON TODOS LOS REQUERIMIENTOS, MANEJA EL VEHÍCULO CON SEGURIDAD. SI ALGUNA DE TUS RESPUESTA FUE NO, CORRÍGELO EN COORDINACION CON TU JEFE INMEDIATO Y CON LA OFICINA DE SERVICIOS GENERALES";
      const textoDividido = doc.splitTextToSize(notaAviso, pageWidth - (margin * 2));
      doc.text(textoDividido, margin, pageHeight - 20); // Justo arriba del pie de página

      // 6. FIRMA, NOMBRE Y RPE
      // Vamos a la última página creada
      const totalPages = doc.internal.getNumberOfPages();
      doc.setPage(totalPages);
      
      if (yPos + 50 > pageHeight - 30) { doc.addPage(); yPos = 20; }
      
      doc.setFontSize(10);
      doc.setFont('helvetica', 'bold');
      doc.text("FIRMA DEL TRABAJADOR:", margin, yPos);
      yPos += 5;
      
      if (datos.firma) {
        doc.addImage(datos.firma, 'PNG', margin, yPos, 60, 30);
      }
      yPos += 30;
      doc.line(margin, yPos, margin + 70, yPos);
      yPos += 5;
      // Aquí agregamos los datos recogidos en el Paso 1
      doc.setFont('helvetica', 'normal');
      doc.text(`Nombre: ${f.nombre || 'No especificado'}`, margin, yPos);
      yPos += 5;
      doc.text(`RPE: ${f.rpe || 'No especificado'}`, margin, yPos);

      // 7. FOTO O EVIDENCIA 
      if (datos.imagen) {
        doc.addPage(); // La foto oficial va en una hoja anexa
        doc.setFontSize(12);
        doc.setFont('helvetica', 'bold');
        doc.text("ANEXO: EVIDENCIA FOTOGRÁFICA (FALLOS DETECTADOS)", margin, 20);
        
        doc.setFontSize(9);
        doc.setFont('helvetica', 'normal');
        doc.text(`Ubicación GPS: ${datos.direccion || ''}`, margin, 30);
        if(datos.coordenadas){
           doc.text(`Coordenadas: ${datos.coordenadas.latitude}, ${datos.coordenadas.longitude}`, margin, 35);
        }
        doc.addImage(datos.imagen, 'JPEG', margin, 45, 120, 90);
      }

      doc.save(`Verificacion_${f.noEco || 'Vehiculo'}_${ID_GENERADO}.pdf`);

    } catch (error) {
      console.error('Error generando PDF:', error);
      alert('Error al generar PDF');
    }
  };

  return (
    <div style={{ textAlign: 'center', padding: '20px' }}>
      <h2>Reporte Finalizado</h2>
      <button onClick={generarPDF} style={{ background: '#4CAF50', color: 'white', padding: '15px 30px', border: 'none', borderRadius: '8px', cursor: 'pointer', fontSize: '16px' }}>
        📄 Generar Documento Oficial CFE
      </button>
    </div>
  );
};

export default Reporte;





import React, { useRef, useState } from 'react';

const Ubicacion = ({ onUbicacionObtenida }) => {
  const [cargando, setCargando] = useState(false);
  const [error, setError] = useState(null);

  const obtenerUbicacion = () => {
    setCargando(true);
    setError(null);

    if (!navigator.geolocation) {
      setError('Geolocalización no soportada por el navegador');
      setCargando(false);
      return;
    }

    navigator.geolocation.getCurrentPosition(
      async (position) => {
        const { latitude, longitude } = position.coords;
        const coordenadas = { lat: latitude, lng: longitude };
        
        try {
          // Obtener dirección de OpenStreetMap
          const response = await fetch(
            `https://nominatim.openstreetmap.org/reverse?format=json&lat=${latitude}&lon=${longitude}&zoom=18&addressdetails=1`,
            {
              headers: {
                'Accept-Language': 'es',
                'User-Agent': 'SeguriApp/1.0'
              }
            }
          );
          
          const data = await response.json();
          const direccion = data.display_name || 'Dirección no disponible';
          
          onUbicacionObtenida(
            `Lat: ${latitude.toFixed(6)}, Lon: ${longitude.toFixed(6)}`,
            direccion,
            coordenadas
          );
        } catch (err) {
          console.error('Error obteniendo dirección:', err);
          onUbicacionObtenida(
            `Lat: ${latitude.toFixed(6)}, Lon: ${longitude.toFixed(6)}`,
            'Error al obtener dirección detallada',
            coordenadas
          );
        }
        
        setCargando(false);
      },
      (err) => {
        console.error('Error de geolocalización:', err);
        setError('No se pudo obtener la ubicación. Verifica los permisos.');
        setCargando(false);
      },
      {
        enableHighAccuracy: true,
        timeout: 10000,
        maximumAge: 0
      }
    );
  };

  return (
    <div className="bg-white p-6 rounded-lg shadow-lg">
      <h2 className="text-xl font-bold mb-4">2. Registrar Ubicación</h2>
      
      <div className="mb-6">
        <p className="text-gray-600 mb-4">
          Para registrar la ubicación exacta del incidente, necesitamos tu permiso.
        </p>
        
        <button
          onClick={obtenerUbicacion}
          disabled={cargando}
          className={`w-full py-3 rounded-lg font-medium ${
            cargando 
              ? 'bg-gray-400 cursor-not-allowed' 
              : 'bg-green-600 hover:bg-green-700 text-white'
          }`}
        >
          {cargando ? (
            <span className="flex items-center justify-center">
              <svg className="animate-spin h-5 w-5 mr-3 text-white" viewBox="0 0 24 24">
                <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" fill="none" />
                <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z" />
              </svg>
              Obteniendo ubicación...
            </span>
          ) : (
            '📍 Obtener Ubicación Actual'
          )}
        </button>
      </div>

      {error && (
        <div className="bg-red-50 border border-red-200 text-red-700 px-4 py-3 rounded-lg mb-4">
          <strong>Error:</strong> {error}
        </div>
      )}

      <div className="bg-blue-50 border border-blue-200 p-4 rounded-lg">
        <h3 className="font-bold text-blue-800 mb-2">¿Por qué necesitamos tu ubicación?</h3>
        <ul className="text-blue-700 text-sm space-y-1">
          <li>• Registro exacto del lugar del incidente</li>
          <li>• Cumplimiento de requisitos legales</li>
          <li>• Posible envío de asistencia</li>
          <li>• Evidencia georreferenciada</li>
        </ul>
      </div>
    </div>
  );
};

export default Ubicacion;




import React, { useState } from 'react';
import DatosGenerales from './components/DatosGenerales'; // <-- NUEVO
import Checklist, { estructura } from './components/Checklist';
import Camara from './components/Camara';
import Ubicacion from './components/Ubicacion';
import Firma from './components/Firma';
import Reporte from './components/Reporte';

function App() {
  const [paso, setPaso] = useState(1); // Ahora el paso 1 son los Datos
  const [checklist, setChecklist] = useState({});
  const [datosForm, setDatosForm] = useState({}); // <-- Estado para el nuevo formulario

  const [datos, setDatos] = useState({
    imagen: null,
    ubicacion: null,
    direccion: '',
    hora: '',
    firma: null,
    coordenadas: null
  });

  const actualizarDatos = (nuevos) => setDatos(prev => ({ ...prev, ...nuevos }));
  const actualizarDatosForm = (nuevos) => setDatosForm(prev => ({ ...prev, ...nuevos }));

  const siguientePaso = () => {
    // -------------------------------------------------
    // 1. VALIDACIÓN DATOS GENERALES (NUEVO PASO 1)
    // -------------------------------------------------
    if (paso === 1) {
      const camposObligatorios = ['noEco', 'area', 'periodo', 'mes', 'anio', 'kilometraje', 'nombre', 'rpe'];
      const faltan = camposObligatorios.filter(campo => !datosForm[campo] || datosForm[campo].trim() === '');
      
      if (faltan.length > 0) {
        alert(`⚠️ FALTAN DATOS.\nPor favor llena todos los campos del formulario.`);
        return;
      }
    }

    // -------------------------------------------------
    // 2. VALIDACIÓN DEL CHECKLIST (AHORA PASO 2)
    // -------------------------------------------------
    if (paso === 2) {
      const itemsTotales = Object.values(estructura).flat();
      const faltantes = itemsTotales.filter(item => !checklist[item]);

      if (faltantes.length > 0) {
        alert(`⚠️ NO PUEDES AVANZAR.\nFaltan por contestar:\n- ${faltantes.join("\n- ")}`);
        return; 
      }

      const hayObservaciones = Object.values(checklist).some(val => val === "NO" || val === "NA");
      if (!hayObservaciones) {
        // Todo OK, saltamos la cámara y vamos a la ubicación (Paso 4)
        actualizarDatos({ imagen: null }); 
        setPaso(4); 
        return;
      }
    }

    // 3. FOTO OBLIGATORIA
    if (paso === 3 && !datos.imagen) {
      alert("⚠️ FOTO OBLIGATORIA.\nComo reportaste fallos, debes tomar foto.");
      return;
    }

    // 4. UBICACIÓN OBLIGATORIA
    if (paso === 4 && !datos.ubicacion) {
      alert("⚠️ UBICACIÓN REQUERIDA.\nRegistra tu ubicación GPS para continuar.");
      return;
    }

    // 5. FIRMA OBLIGATORIA
    if (paso === 5 && !datos.firma) {
      alert("⚠️ FIRMA FALTANTE.\nDebes guardar tu firma para continuar.");
      return;
    }

    if (paso < 6) setPaso(p => p + 1);
  };

  const anteriorPaso = () => {
    if (paso === 2) setPaso(1); // De checklist a datos
    else if (paso === 3) setPaso(2); // De cámara a checklist
    else if (paso === 4) {
       // Si estamos en ubicación, checamos si saltamos la cámara
       const hayObservaciones = Object.values(checklist).some(val => val === "NO" || val === "NA");
       if (!hayObservaciones) setPaso(2); // Regresa al checklist
       else setPaso(3); // Regresa a cámara
    }
  };

  const guardarInspeccion = () => {
    const historial = JSON.parse(localStorage.getItem("inspecciones")) || [];
    const nueva = { id: crypto.randomUUID(), fecha: new Date().toISOString(), checklist, datosForm, ...datos };
    historial.push(nueva);
    localStorage.setItem("inspecciones", JSON.stringify(historial));
  };

  return (
    <div className="app">
      <header>
        <h1>SeguridApp</h1>
        <p className="subtitulo">Registro de Inspección CFE</p>
        <div className="progreso-container">
          <div className="progreso">
            <div className={`paso ${paso >= 1 ? 'activo' : ''}`}>1. Datos</div>
            <div className={`paso ${paso >= 2 ? 'activo' : ''}`}>2. Lista</div>
            <div className={`paso ${paso >= 3 ? 'activo' : ''} ${paso === 4 && !datos.imagen && paso > 2 ? 'saltado' : ''}`}>3. Foto</div>
            <div className={`paso ${paso >= 4 ? 'activo' : ''}`}>4. Ubic.</div>
            <div className={`paso ${paso >= 5 ? 'activo' : ''}`}>5. Firma</div>
            <div className={`paso ${paso >= 6 ? 'activo' : ''}`}>6. Fin</div>
          </div>
        </div>
      </header>

      <main className="contenido">
{paso === 1 && (
          <DatosGenerales 
            datos={datosForm} 
            onChange={actualizarDatosForm} 
            onCargarDatos={(datosExtraidos) => {
              // Esta función recibe los datos del PDF y los inyecta en la app
              setDatosForm(datosExtraidos.form || {});
              setChecklist(datosExtraidos.checklist || {});
            }}
          />
        )}        {paso === 2 && <Checklist onChange={setChecklist} datosPrevios={checklist} />}
        {paso === 3 && (
          <div className="fade-in">
            <div className="alerta-foto">📸 Evidencia Obligatoria por reporte de fallos.</div>
            <Camara onCapture={(imagen, hora) => { actualizarDatos({ imagen, hora }); }} />
          </div>
        )}
        {paso === 4 && <Ubicacion onUbicacionObtenida={(ubicacion, direccion, coordenadas) => actualizarDatos({ ubicacion, direccion, coordenadas })} />}
        {paso === 5 && <Firma onFirmaCompletada={(firma) => actualizarDatos({ firma })} />}
        {paso === 6 && <Reporte datos={{ ...datos, checklist, form: datosForm }} onGuardar={guardarInspeccion} />}

        <div className="navegacion">
          {(paso === 2 || paso === 3 || (paso === 4 && !datos.imagen)) && (
            <button onClick={anteriorPaso} className="btn btn-secundario">← Regresar</button>
          )}
          {paso !== 2 && paso !== 3 && !(paso === 4 && !datos.imagen) && <div></div>}
          {paso < 6 && (
            <button onClick={siguientePaso} className="btn btn-principal">Siguiente →</button>
          )}
        </div>
      </main>

      <style>{`
        * { box-sizing: border-box; }
        .app { padding: 15px; max-width: 800px; margin: 0 auto; font-family: sans-serif; background: #f5f7fa; min-height: 100vh;}
        header { text-align: center; margin-bottom: 20px; }
        h1 { margin: 0; color: #1a365d; font-size: 1.8rem; }
        .subtitulo { margin: 5px 0 15px 0; color: #718096; font-size: 0.9rem; }
        .progreso-container { overflow-x: auto; }
        .progreso { display: flex; gap: 4px; background: #e2e8f0; padding: 4px; border-radius: 8px; }
        .paso { flex: 1; padding: 6px 2px; text-align: center; border-radius: 6px; font-size: 0.7rem; color: #718096; background: white; white-space: nowrap; }
        .paso.activo { background: #2196F3; color: white; font-weight: bold; }
        .paso.saltado { background: #90CDF4; color: white; opacity: 0.6; }
        .contenido { background: white; padding: 20px; border-radius: 12px; box-shadow: 0 4px 6px rgba(0,0,0,0.05); margin-bottom: 20px; }
        .alerta-foto { background: #fff3cd; color: #856404; padding: 10px; border-radius: 6px; margin-bottom: 15px; border-left: 4px solid #ffeeba; }
        .navegacion { display: flex; justify-content: space-between; gap: 15px; margin-top: 25px; }
        .btn { padding: 12px 15px; border: none; border-radius: 8px; cursor: pointer; font-size: 1rem; font-weight: 600; flex: 1; transition: 0.2s;}
        .btn-principal { background: #2196F3; color: white; }
        .btn-secundario { background: #CBD5E0; color: #4A5568; }
        .fade-in { animation: fadeIn 0.4s ease-in; }
        @keyframes fadeIn { from { opacity: 0; } to { opacity: 1; } }
        @media (max-width: 600px) { .navegacion { flex-direction: column-reverse; } .btn { width: 100%; } }
      `}</style>
    </div>
  );
}
export default App;
